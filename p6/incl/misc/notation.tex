\textbf{Abstract Algebra}

\begin{table}[H]
    \begin{tabular}{ll}
      $\F_{q}$ & The finite field with $q = p^{n}$ elements. \\
      $\cF$ & A closed field. \\
      $\pols$ & The multivariate polynomial ring over $\cF$, also written as $\cF[x_{1}, \ldots, x_{n}]$. \\
      $\gen{S}$ & The ideal generated by $S$.
    \end{tabular}
\end{table}

\textbf{Algebraic Geometry}
\begin{table}[H]
    \begin{tabular}{ll}
      $\A^{n}$ & The $n$ dimensional affine space. \\
      $Z(S)$ & The zero set of $S \subseteq \pols$. \\
      $\Zar$ & The Zariski topology. \\
      $\cZar$ & The closed sets in the Zariski topology (the affine algebraic sets).
    \end{tabular}
\end{table}

\textbf{Coding theory}
\begin{table}[H]
    \begin{tabular}{ll}
      $\C$ & A linear code. \\
      $G$ & A generator matrix. \\
      $H$ & A parity check matrix. \\
      $\d$ & The Hamming metric. \\
      $\wt$ & The Hamming weight.
    \end{tabular}
\end{table}
