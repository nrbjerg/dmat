\textbf{Abstract Algebra}

\begin{table}[H]
    \begin{tabular}{ll}
      $\F_{q}$ & The finite field with $q$ elements. \\
      $R[X_1, X_2, \ldots, X_{n}]$ & The multivariate polynomial ring over $R$. \\
      $R^{*}$ & The set of units in $R$. \\
      $\gen{S}$ & The ideal generated by the set $S$. \\
      $\mathbb{K}/\mathbb{F}$ & A field extension, meaning $\mathbb{F}$ is  a subfield of $\mathbb{K}$. \\
      $\cF$ & The algebraic closure of $\mathbb{F}$.
    \end{tabular}
\end{table}

\textbf{Algebraic Geometry}
\begin{table}[H]
    \begin{tabular}{ll}
      $\K$ & An arbitary algebraicly closed field. \\
      $\A^{n}$ & The $n$ dimensional affine space. \\
      $Z(S)$ & The zero set of $S \subseteq \K[X_1, X_2, \ldots, X_{n}]$. \\
      $I(X)$ & The vanishing ideal of $X \subseteq \mathbb{A}^{n}$ \\
      $\Zar$ & The Zariski topology. \\
      $\mathbb{P}^{n}$ & The $n$ dimensional projective space. \\
      $I_{\mathbb{P}}(X)$ & The projective vanishing ideal of $X \subseteq \mathbb{P}^{n}$ \\
      $Z_{\mathbb{P}}(S)$ & The projective zero set of $S \subseteq \K[X_1, X_2, \ldots, X_{n}]$. \\
      $\mathcal{X}$ & An affine or projective variety. \\
      $\K[\mathcal{X}]$ & The cordinate ring over the variety $\mathcal{X}$. \\
      $\K(\mathcal{X})$ & The function field over $\mathcal{X}$. \\
      $\mathcal{O}_{p}(\mathcal{X})$ & The local ring of $\mathcal{X}$ at $p$.
    \end{tabular}
\end{table}

\textbf{Coding theory}
\begin{table}[H]
    \begin{tabular}{ll}
      $\C$ & A linear code. \\
      $G$ & A generator matrix. \\
      $H$ & A parity check matrix. \\
      $\d$ & The Hamming metric. \\
      $\wt$ & The Hamming weight.
    \end{tabular}
\end{table}
